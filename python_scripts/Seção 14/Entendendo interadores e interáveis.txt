"""
Entendendo iterators e iterables

Iterators:
- Um objeto que pode ser iterado;
- Um objeto que retorna um dado, sendo um elemento or vez quando uma função next() é chamada.
Iterables:
- Um objeto que retornará um iterator quando a função iter() for chamada.


nome = 'Luccas'  # É um iterable mas não é um iterator.
numeros = [1, 2, 3, 4, 5] # É um iterable mas não é um iterator.

it1 = iter(nome)
it2 = iter(numeros)

print(next(it1))
print(next(it1))
print(next(it1))
print(next(it1))
print(next(it1))
print(next(it1))

print(next(it2))
print(next(it2))
print(next(it2))
print(next(it2))
print(next(it2))
"""
