"""
len, abs, sum e round

len() -> retorna o tamanho de um iterável
abs() -> Retorna o valor absoluto de um número inteiro ou real. De forma básica, seria seu valor real sem o sinal.
## exemplo:
print(abs(-5))
print(abs(3.14))

# Sum
sum() -> Recebe como parâmetro o iterável, podendo receber um valor inicial, e retorna a soma total dos elementos.
print(sum([1, 4, 3, 1, 1]));

# Round
round() -> Retorna um número arredondado para n digito de precisão após a casa decimal. Se a precisão não for 
informada retorna o inteiro mais próximo da entrada.
print(round(3.14)) # 3.
print(round(3.145872, 2)); # 3.14;
print(round(12.33, 1)); # 12.3;
"""
